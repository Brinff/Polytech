#include <math.h>
#include <locale>

	{

		{
			float grad, radian;
			cout << "\nДля того, чтобы перевести, введите градусы (число от 0 до 360): ";

			radian = grad / 180;
			cout << "\nВаш ответ: " << radian << " pi";
			break;

		}

		case 2:
		{
			float grad, radian;
			cout << "\nДля того, чтобы перевести, введите радианы (число от 0 до 2): ";
			cin >> radian;
			grad = radian * 180;
			cout << "\nВаш ответ: " << grad << " градусов";
			break;

		}

		case 3:
		{
			int x, A;

			cout << "\nКилограмм: ";
