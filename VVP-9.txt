#include <iostream>

{

Определить номер дня недели для K-го дня года, если известно, что в этом году 1 января было днем недели с номером N\n4.Даны целые положительные числа A, B, C. На прямоугольнике размера A x B размещено максимально возможное количество квадратов со стороной C (без наложений). Найти количество квадратов, размещенных на прямоугольнике, а также площадь незанятой части прямоугольника.\n5.Дан номер некоторого года (целое положительное число). Определить соответствующий ему номер столетия.\n";
	cin >> various;

	{
		case 1:
		{
			int sec, sec1;
			cout << "\nВведите кол-во секунд, прошедших с начала дня, чтобы определить кол-во секунд, прошедших с начала последней минуты: ";
			cin >> sec;
