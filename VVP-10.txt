#include <iostream>
#include <locale>
using namespace std;

int main()
{
	int various;
	setlocale(LC_ALL, "Rus");
	cout << "\nПроверка истинности выражения.\nЧтобы проверить выражение на истинность, выберите его:\n1.Справедливы неравенства A > 2 и B <= 3\n2.Справедливо двойное неравенство A < B < C\n3.Данное число является четным двузначным\n4.Цифры данного числа образуют возрастающую или убывающую последовательность\n5.Данное число читается одинаково слева направо и справа налево\n6.Треугольник со сторонами a, b, c является прямоугольным\n7.Существует треугольник со сторонами a, b, c\n";
	cin >> various;

	switch (various)
	{
		case 1:
		{
			int a, b;
			cout << "\nЧтобы проверить истинность (A > 2 и B <= 3), введите А и В";

			cout << "\nВведите А: ";
			cin >> a;

			cout << "\nВведите B: ";
			cin >> b;

			if (a > 2 && b <= 3)
				cout << "\nВыражение истинно";
			else
				cout << "\nВыражение ложно";
			break;


		}

		case 2:
		{
			int a, b, c;
			cout << "\nЧтобы проверить истинность (A < B < C), введите А, B и C";

			cout << "\nВведите А: ";
			cin >> a;

			cout << "\nВведите B: ";
			cin >> b;

			cout << "\nВведите C: ";
			cin >> c;

			if (a < b && a < c && b < c)
				cout << "\nВыражение истинно";
			else
				cout << "\nВыражение ложно";
			break;


		}

		case 3:
		{
			int a;
			cout << "\nЧтобы проверить истинно ли выражение (данное число является четным двузначным), введите число: ";
			cin >> a;

			if (a < 100 && a > 9 && a % 2 == 0)
				cout << "\nВыражение истинно";
			else
				cout << "\nВыражение ложно";
			break;

		}

		case 4:
		{
			int n;
			cout << "\nЧтобы проверить истинно ли выражение (Цифры данного числа образуют возрастающую или убывающую последовательность), введите трехзначное число: ";
			cin >> n;

			if ((n / 100 > (n % 100) / 10 && n / 100 > n % 10 && (n % 100) / 10 > n % 10) || (n / 100 < (n % 100) / 10 && n / 100 < n % 10 && (n % 100) / 10 < n % 10))
				cout << "\nВыражение истинно";
			else
				cout << "\nВыражение ложно";
			break;
		}

		case 5:
		{
			int n;
			cout << "\nЧтобы проверить истинно ли выражение (Данное число читается одинаково слева направо и справа налево), введите четырехзначное число: ";
			cin >> n;

			if (n / 1000 == n % 10 && (n % 1000) / 100 == (n % 100) / 10 )
				cout << "\nВыражение истинно";
			else
				cout << "\nВыражение ложно";
			break;
		}

		case 6:
		{
			int a, b, c;
			cout << "\nЧтобы проверить истинно ли выражение (Треугольник со сторонами a, b, c является прямоугольным), введите стороны треугольника abc: ";

			cout << "\nВведите А: ";
			cin >> a;

			cout << "\nВведите B: ";
			cin >> b;

			cout << "\nВведите C: ";
			cin >> c;

			if (a * a == b * b + c * c || b * b == a * a + c * c || c * c == a * a + b * b)
				cout << "\nВыражение истинно";
			else
				cout << "\nВыражение ложно";
			break;

		}

		case 7:
		{
			int a, b, c;
			cout << "\nЧтобы проверить истинно ли выражение (Существует треугольник со сторонами a, b, c), введите стороны треугольника abc: ";

			cout << "\nВведите А: ";
			cin >> a;

			cout << "\nВведите B: ";
			cin >> b;

			cout << "\nВведите C: ";
			cin >> c;

			if (a > b && a > c && a < b + c)
				cout << "\nВыражение истинно";
			else if (b > a && b > c && b < a + c)
				cout << "\nВыражение истинно";
			else if (c > b && c > a && c < a + b)
				cout << "\nВыражение истинно";
			else if (a == b == c)
				cout << "\nВыражение истинно";
			else
				cout << "\nВыражение ложно";
			break;

		}

		default:
			cout << "\nВы ввели число не от 1 до 7.";
			break;
	}
	return 0;
}
