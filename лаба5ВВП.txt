#include <iostream>
#include <math.h>
#include <locale>
using namespace std;

int main()
{
	setlocale(LC_ALL, "Rus");

	int various;

	cout << "\nВыберите необходимую операцию из списка:\n1.Найти расстояние между двумя точками с заданными координатами.\n2.Найти длины отрезков AC и BC и их сумму.\n3.Найти произведение длин отрезков AC и BC.\n4.Найти периметр и площадь прямоугольника, по данным точкам противоположных вершин.\n5.Найти периметр и площадь треугольника по координатам трех его вершин.\n";
	
	cin >> various;

	switch (various)
	{
		case 1:
		{
			int x1 = 0, x2 = 0, y1 = 0, y2 = 0;

			cout << "\nДля того, чтобы найти расстояние между двумя точками с заданными координатами, введите x1, y1, x2, y2:  ";
			cout << "\nx1, y1: \n";

			cin >> x1;
			cin >> y1;

			cout << "\nx2, y2: \n";

			cin >> x2;
			cin >> y2;



			cout << "\nВаш результат: \nРасстояние: " << sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));


			break;
		}

		case 2:
		{
			int A = 0, B = 0, C = 0;
			cout << "\nДля того, чтобы найти длины отрезков AC и BC и их сумму, введите A, B, C, в порядке возрастания:  ";

			cout << "\nA: ";
			cin >> A;

			cout << "\nB: ";
			cin >> B;

			cout << "\nC: ";
			cin >> C;

			if (C > B and C > A and B > A)
			{
				cout << "\nВаш результат: ";
				cout << "\nДлинна AC: " << C - A;
				cout << "\nДлинна BC: " << C - B;
				cout << "\nСумма отрезков: " << (C - A) + (C - B);
			}

			else
			{
				cout << "\nОшибка. Вы ввели данные не в порядке возрастания.";
			}

			break;
		}

		case 3:
		{
			int A = 0, B = 0, C = 0;
			cout << "\nДля того, чтобы найти произведение длин отрезков AC и BC, введите A, C, B, в порядке возрастания:  ";

			cout << "\nA: ";
			cin >> A;

			cout << "\nC: ";
			cin >> C;

			cout << "\nB: ";
			cin >> B;

			if (B > C and B > A and C > A)
			{
				cout << "\nВаш результат: ";
				cout << "\nДлинна AC: " << C - A;
				cout << "\nДлинна BC: " << B - C;
				cout << "\nПроизведение длин: " << (C - A) * (B - C);
			}

			else
			{
				cout << "\nОшибка. Вы ввели данные не в порядке возрастания.";
			}

			break;
		}

		case 4:
		{
			int x1 = 0, x2 = 0, y1 = 0, y2 = 0;

			cout << "\nx1, y1: \n";

			cin >> y1;

			cout << "\nx2, y2: \n";

			cin >> x2;
			cin >> y2;

			cout << "\nПлощадь: " << abs(y1 - y2) * abs(x1 - x2);


			

			break;
		}


		case 5:
		{
			int x1 = 0, x2 = 0, y1 = 0, y2 = 0, x3 = 0, y3 = 0;

			cout << "\nДля того, чтобы найти периметр и площадь треугольника по координатам его вершин, введите x1, y1, x2, y2, x3, y3:  ";

			cout << "\nx1, y1: \n";

			cin >> y1;

			cin >> y2;

			cin >> x3;
			cin >> y3;

			cout << "\nВаш результат: \nПлощадь: " << (1 / 2) * abs((x2 - x1) * (y3 - y1) - (x3 - x1) * (y2 - y1));
			cout << "\nПериметр: " << sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1)) + sqrt((x3 - x1) * (x3 - x1) + (y3 - y1) * (y3 - y1)) + sqrt((x3 - x2) * (x3 - x2) + (y3 - y2) * (y3 - y2));

			break;
		}

		default:
			cout << "Вы ввели не цифру от 1 до 5. Ошибка.";
			break;
	}
	return 0;
}
