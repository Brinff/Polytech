#include <iostream>
#include <math.h>
#include <locale>
 
using namespace std;

int main()
{
	setlocale(LC_ALL, "Rus");

	int various;

	cout << "\nВыберите необходимую операцию из списка:\n1.Поменять местами содержимое переменных A и B и вывести новые значения A и B.\n2.Изменить значения, переместив содержимое A в B, B — в C, C — в A, и вывести новые значения переменных A, B, C.\n3.Изменить значения, переместив содержимое A в C, C — в B, B — в A, и вывести новые значения переменных A, B, C.\n4.Найти значение функции y = 3x6 − 6x2 − 7 при данном значении x.\n5.Найти значение функции y = 4(x−3)6 − 7(x−3)3 + 2 при данном значении x.\n6.Дано число A. Вычислить A^8 , используя вспомогательную переменную и три операции умножения.\n7.Дано число A. Вычислить A^15, используя две вспомогательные переменные и пять операций умножения.\n";

	cin >> various;

	switch (various)
	{
	case 1:
	{
		int A, B;

		cout << "\nДля того, чтобы поменять переменные местами, введите их:  ";
		cout << "\nА: \n";

		cin >> A;
		

		cout << "\nB: \n";

		cin >> B;
		
		A = A ^ B;
		B = B ^ A;
		A = A ^ B;



		cout << "\nВаш результат: ";
		cout << "\nA = " << A;
		cout << "\nB = " << B;


		break;
	}

	case 2:
	{
		int A = 0, B = 0, C = 0, a = 0, b = 0, c = 0;
		cout << "\nДля того, чтобы поменять значения местами, введите A, B, C:  ";

		cout << "\nA: ";
		cin >> A;

		cout << "\nB: ";
		cin >> B;

		cout << "\nC: ";
		cin >> C;

		a = A;
		b = B;
		c = C;

		B = a;
		C = b;
		A = c;

		cout << "\nВаши результаты: ";
		cout << "\nA = " << A;
		cout << "\nB = " << B;
		cout << "\nC = " << C;

		break;
	}

	case 3:
	{
		int A = 0, B = 0, C = 0, a = 0, b = 0, c = 0;
		cout << "\nДля того, чтобы поменять значения местами, введите A, B, C:  ";

		cout << "\nA: ";
		cin >> A;

		cout << "\nB: ";
		cin >> B;

		cout << "\nC: ";
		cin >> C;

		a = A;
		b = B;
		c = C;

		B = c;
		C = a;
		A = b;

		cout << "\nВаши результаты: ";
		cout << "\nA = " << A;
		cout << "\nB = " << B;
		cout << "\nC = " << C;

		break;
	}

	case 4:
	{
		int x, y;

		cout << "\nДля того, чтобы найти значение выражения, введите x:  ";

		cin >> x;
		y = 3 * pow(x, 6) - 6 * x * x - 7;

		cout << "Ваш результат, у = " << y;
		
		break;
	}


	case 5:
	{
		int x, y;

		cout << "\nДля того, чтобы найти значение выражения, введите x:  ";

		cin >> x;
		y = 4 * pow(x - 3, 6) - 7 * pow(x - 3, 3) + 2;

		cout << "Ваш результат, у = " << y;

		break;
	}

	case 6:
	{
		int A, a, res;

		cout << "Чтобы вывести А^8, введите А: ";
		cin >> A;

		a = pow(A, 2);
		res = a * a * a * a;

		cout << "Результат, A^8 = " << res;

		break;
	}

	case 7:
	{
		int A, a, b, res;

		cout << "Чтобы вывести А^15, введите А: ";
		cin >> A;

		a = pow(A, 2);
		b = pow(A, 3);
		res = a * b * a * b * a * b;

		cout << "Результат, A^15 = " << res;

		break;
	}

	default:
		cout << "Вы ввели не цифру от 1 до 7. Ошибка.";
		break;
	}
	return 0;
}
